Reusable form input with Angular

Demo

Code used

index.html

<form name="myForm" novalidate>
    <form-input></form-input>
    <button type="submit" ng-disabled="myForm.name.$invalid">submit</button>
</form>

input-field.html

<div ng-repeat="data in $ctrl.inputs">
  <label>{{data.label}}</label>
  <input type="{{data.type}}" name="name" ng-model="name" placeholder="{{data.placeholder}}" required />
</div>

angular.js

angular.module('myApp', [
'formInput'
]);

angular.module('formInput', []);

angular.module('formInput').component('formInput', {
	
    templateUrl: 'input-field.html',
    controller: ['$http',
      function formInputController($http) {
        var self = this;

        $http.get('inputs.json').then(function(response) {
          self.inputs = response.data;
        });
      }
    ]
	
});
